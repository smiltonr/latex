\begin{frame}
  \frametitle{Stepwise revelation}
  \begin{columns}
    \begin{column}{.5\textwidth}
      \begin{block}{Just a block}
        \begin{itemize}
        \item<1-> First step.
        \item<2-> Second step.
        \item<3-> Third step.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{.5\textwidth}
      \begin{alertblock}{Another block}
        \begin{itemize}
        \item<1-> A step.
        \item<2-> B step.
        \item<3-> C step.
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}
